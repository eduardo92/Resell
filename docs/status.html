Database Schema ‚Äî Sites Table

CREATE TABLE Sites (
  slug NVARCHAR(255) PRIMARY KEY,
  businessName NVARCHAR(500) NOT NULL,
  status NVARCHAR(50) NOT NULL, -- nuevo | generando | listo | contactado | vendido
  configJson NVARCHAR(MAX) NOT NULL, -- full SiteConfig JSON blob
  lastUpdated DATETIME2 DEFAULT GETDATE()
)
